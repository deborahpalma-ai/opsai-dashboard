Connect your database

OpsAI uses Supabase to store all data securely in the cloud. Free, no credit card required. Follow the 3 steps below once.

1
Create a free Supabase account
Go to supabase.com, sign up (GitHub or Google), create a new project and wait ~2 min for it to be ready.
2
Run this SQL in your project
Supabase dashboard → SQL Editor → New query → paste and run:
create table if not exists public.profiles(id uuid references auth.users on delete cascade primary key,email text,name text default '',company text default '',workspace text default '',lang text default 'en',theme text default 'light',onboarding_done boolean default false,ai_prio_ct int default 0,sessions int default 7,created_at timestamptz default now()); alter table public.profiles enable row level security; create policy "own profile" on public.profiles for all using(auth.uid()=id); create table if not exists public.tasks(id text primary key default gen_random_uuid()::text,user_id uuid references auth.users on delete cascade not null,name text not null,prio text default 'm',date text default 'this-week',done boolean default false,created_at timestamptz default now()); alter table public.tasks enable row level security; create policy "own tasks" on public.tasks for all using(auth.uid()=user_id); create table if not exists public.meetings(id text primary key default gen_random_uuid()::text,user_id uuid references auth.users on delete cascade not null,title text default 'Meeting',transcript text default '',summary text default '',created_at timestamptz default now()); alter table public.meetings enable row level security; create policy "own meetings" on public.meetings for all using(auth.uid()=user_id); create table if not exists public.shared_settings(key text primary key,value text default ''); alter table public.shared_settings enable row level security; create policy "read shared" on public.shared_settings for select using(true); create policy "write shared" on public.shared_settings for all using(true); create table if not exists public.weekly_stats(user_id uuid references auth.users on delete cascade,day_index int,count int default 0,primary key(user_id,day_index)); alter table public.weekly_stats enable row level security; create policy "own stats" on public.weekly_stats for all using(auth.uid()=user_id);
3
Get your Project URL and anon key
Supabase → Settings → API. Copy Project URL and anon public key below.
